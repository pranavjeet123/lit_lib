üìù Usage Examples

HTML

<!-- Basic usage -->
<rating-component value="3"></rating-component>

<!-- With custom attributes -->
<rating-component 
  value="4" 
  max-rating="5" 
  size="large"
  show-label="true"
></rating-component>

<!-- Read-only display -->
<rating-component value="4.5" readonly></rating-component>

JavaScript / TypeScript

// Listen to rating changes
const rating = document.querySelector('rating-component');
rating.addEventListener('rating-change', (e) => {
  console.log('New rating:', e.detail.value);
  console.log('Old rating:', e.detail.oldValue);
});

// Set value programmatically
rating.value = 4;

// Get current value
console.log(rating.value);

NPM Installation

npm install @your-username/rating-component
